12. post '/api/favorites/update'
    const { msisdn, status, trackId } = body
    if (status === 1) {
        await db.addFavorites(msisdn, trackId)
    } else if (status === -1) {
        await db.removeFavorites(msisdn, trackId)
    }
    return { status: 'ok' }
